let wnd = api.window();

wnd.create(600, 400);

let dragCrt = func() {
	if (wnd.isAlive() == 0) {
		print("Window closed!");
		wnd = none;
		return;
	}
	wnd.drag();
	timeout(dragCrt, 0);
};

let device = api.device();
try {
	device.init(wnd);
	print("Init Complete!");
} catch(e) {
	print(e);
}


dragCrt();
