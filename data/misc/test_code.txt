if (1 == 1) {
	print("Equal Works");
}

if (1 < 2) {
	print("Less Works");
}

if (1 > 0) {
	print("Greater Works");
}

if (1 <= 1) {
	print("LEQ Works");
}

if (1 >= 1) {
	print("GEQ Works");
}

if (1 >= 1 && 1 == 1) {
	print("Conjunction Works");
}

if (1 >= 1 || 1 == 1) {
	print("Disjunction Works");
}
